const ACCESS_PASSWORD = "MHP1002025"; // cámbiala a la que quieras const passInput = document.querySelector("#passInput"); const msg = document.querySelector("#msg"); const menu = document.querySelector("#menu"); function checkPassword(){ const value = (passInput.value || "").trim(); if(!value){ msg.textContent = "Ingresa la contraseña."; msg.className = "msg err"; menu.style.display = "none"; return; } if(value === ACCESS_PASSWORD){ msg.textContent = "Acceso concedido."; msg.className = "msg ok"; menu.style.display = "block"; }else{ msg.textContent = "Contraseña incorrecta."; msg.className = "msg err"; menu.style.display = "none"; } }
